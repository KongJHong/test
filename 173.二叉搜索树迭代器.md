## 二叉搜索树迭代器

### 问题描述

实现一个二叉搜索树迭代器。你将使用二叉搜索树的根节点初始化迭代器。

调用 `next()` 将返回二叉搜索树中的下一个最小的数。

 

**示例：**

**![img](https://assets.leetcode-cn.com/aliyun-lc-upload/uploads/2018/12/25/bst-tree.png)**

```
BSTIterator iterator = new BSTIterator(root);
iterator.next();    // 返回 3
iterator.next();    // 返回 7
iterator.hasNext(); // 返回 true
iterator.next();    // 返回 9
iterator.hasNext(); // 返回 true
iterator.next();    // 返回 15
iterator.hasNext(); // 返回 true
iterator.next();    // 返回 20
iterator.hasNext(); // 返回 false
```

### 思路

中序遍历，栈实现

利用栈的特性，一条条分支进行遍历

```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class BSTIterator {
private:
    stack<TreeNode*> _stack;
public:
    BSTIterator(TreeNode* root) {
        TreeNode* p = root;
        while(p){
            _stack.push(p);
            p = p->left;
        }
    }
    
    /** @return the next smallest number */
    int next() {
        TreeNode *p = _stack.top();
        _stack.pop();
        TreeNode *cp = p->right;
        while(cp){
            _stack.push(cp);
            cp = cp->left;
        }
        
        return p->val;
    }
    
    /** @return whether we have a next smallest number */
    bool hasNext() {
        return !_stack.empty();
    }
};

/**
 * Your BSTIterator object will be instantiated and called as such:
 * BSTIterator* obj = new BSTIterator(root);
 * int param_1 = obj->next();
 * bool param_2 = obj->hasNext();
 */
```

```go
/**
 * Definition for a binary tree node.
 * type TreeNode struct {
 *     Val int
 *     Left *TreeNode
 *     Right *TreeNode
 * }
 */
type BSTIterator struct {
    stack []*TreeNode
    idx int
}


func (bstIter *BSTIterator)Inorder(root *TreeNode){
    if root == nil {
        return
    }
    bstIter.Inorder(root.Left)
    bstIter.stack = append(bstIter.stack,root)
    bstIter.Inorder(root.Right)
}


func Constructor(root *TreeNode) BSTIterator {
    var (
        bstIter *BSTIterator
    )
    
    bstIter = &BSTIterator{
        stack: make([]*TreeNode,0),
        idx:0,
    }
    
    bstIter.Inorder(root)
    
    return *bstIter
}


/** @return the next smallest number */
func (this *BSTIterator) Next() int {
    node := this.stack[this.idx]
    this.idx++
    return node.Val
    
}


/** @return whether we have a next smallest number */
func (this *BSTIterator) HasNext() bool {
    return this.idx != len(this.stack)
}


/**
 * Your BSTIterator object will be instantiated and called as such:
 * obj := Constructor(root);
 * param_1 := obj.Next();
 * param_2 := obj.HasNext();
 */
```
