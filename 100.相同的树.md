给定两个二叉树，编写一个函数来检验它们是否相同。

如果两个树在结构上相同，并且节点具有相同的值，则认为它们是相同的。

```
示例 1:

输入:       1         1
          / \       / \
         2   3     2   3

        [1,2,3],   [1,2,3]

输出: true
示例 2:

输入:      1          1
          /           \
         2             2

        [1,2],     [1,null,2]

输出: false
示例 3:

输入:       1         1
          / \       / \
         2   1     1   2

        [1,2,1],   [1,1,2]

输出: false
```

-------------------

### 思路

就对树分别做dfs，比较返回的数组

```CPP
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
    bool isSameTree(TreeNode* p, TreeNode* q) {
        vector<int> tmp;
        vector<int> pVal;
        vector<int> qVal;
        treeVal(p,pVal);
        treeVal(q,qVal);
        if(pVal != qVal)return false;
        return true;
    }
    
    void treeVal(TreeNode *p,vector<int> &nums){
        if(!p){
            nums.push_back(NULL);
            return;
        }
        nums.push_back(p->val);
        treeVal(p->left,nums);
        treeVal(p->right,nums);
    }
};
```
